with sessions as
(
	select * 
	
	from 
	(
		select session_id,
		incident_id,
		partner_location,
		agent_profile,
		agent_id,
		completed_at,
		ucid,
		agent_name,
		session_rank
		
		from 
		(
			Select distinct s.session_id,
			s.incident_id,
			case when s.agent_centre in ('IENGERIZER_DEHRADUN') then 'IENERGIZER_DEHRADUN'
				 when s.agent_centre in ('TP_MOHALI') then 'TELE_PERFORMANCE_MOHALI'

			s.agent_id,
			date(s.completed_at) as completed_at,
			concat("$",s.ucid) as ucid,

			row_number() over (partition by s.session_id order by s.incident_id) as session_rank

			from bigfoot_external_neo.mp_cs__session_data_new_fact s
			
			where date(completed_at) >= '2025-08-01' 
			and incident_id not in ('')
		)AA
		
		group by session_id,incident_id,partner_location,agent_profile,agent_id,completed_at,ucid,agent_name,session_rank
	
	)xx
	where session_rank = 1

refund as 
(
	Select ref.*
	,case
	when ref_asc_rank_parent =1 and ref_desc_rank_parent = 1 then 'Single_Refund'
	else 'Multiple_Refund' end as mr_refund_flag
  	,case when ref.refund_reason is null then "NA"
			when ref.refund_reason in ('mp_cancellation', 'shield_cancellation', 'seller_cancellation','USER_CANCELLATION', 'cancellation_requested') then "Cancellation"
			when ref.refund_reason in ('undelivered','courier_return','wait_reship') then "RTO"
			when ref.refund_reason in ('customer_return') then "Return"
			when ref.refund_reason in ('nw_adonc_auto_refund') then "Adonc"
			else "Others" end as refund_reason_flag
	,case
	when ref.refund_amount > 10000 then '7. 10K_Plus'
	when ref.refund_amount between 5000 and 10000 then '6. 5k_1k'
	when ref.refund_amount between 3000 and 5000 then '5. 3k_5k'
	when ref.refund_amount between 1500 and 3000 then '4. 1.5k_3k'
	when ref.refund_amount between 1000 and 1500 then '3. 1k_1.5k'
	when ref.refund_amount between 500 and 1000 then '2. 500_1k'
	when ref.refund_amount < 500 then '1. Less_500'
	else cast(ref.refund_amount as string) end as ref_amount_flag

	--,Row_Number() over (partition by ref.order_id, ref.order_item_id order by ref.parent_created_at asc) as ref_asc_rank_parent

	,case 
		when upper(ref.refund_mode) in('CREDIT_CARD_EMI', 'DEBIT_CARD', 'UPI_INTENT', 'CREDIT_CARD', 'PHONEPE', 'NETBANKING', 'UPI_COLLECT', 'IMPS', 'NEFT', 'FK_UPI', 'UPI','SUPERPAY_LATER','SUPERPAY_IN_INSTALLMENT') and  ref.bank_reference_number is NULL then 'ARN_NOT_GENERATED'
		when upper(ref.refund_mode) in('CREDIT_CARD_EMI', 'DEBIT_CARD', 'UPI_INTENT', 'CREDIT_CARD', 'PHONEPE', 'NETBANKING', 'UPI_COLLECT', 'IMPS', 'NEFT', 'FK_UPI', 'UPI','SUPERPAY_LATER','SUPERPAY_IN_INSTALLMENT') and  ref.bank_reference_number is NOT NULL then 'ARN_GENERATED'
		ELSE 'ARN_NOT_APPLICABLE'
		END AS ARN_ISSUE_FLAG

	,case
		when upper(auth_state) = 'VOID' and upper(refund_mode) = 'CREDIT_CARD' then 'void_transaction'
		else 'non_void' end as void_flag
	
	from
	(	
		Select distinct r.user_id
		,r.order_id
		,r.parent_refund_id
		,r.child_refund_id
		,r.parent_created_at
		,r.completed_time
		,r.child_created_at
		,r.child_updated_at
		,r.status as refund_status
		,CASE
		WHEN UPPER(r.status) IN ('COMPLETED') THEN 'Completed'
		WHEN UPPER(r.status) IN ('CLONED', 'FAILED', 'CLONE_REQUESTED') THEN 'Failed'
		WHEN UPPER(r.status) IN ('CANCELLED') THEN 'Cancelled'
		WHEN UPPER(r.status) IN ('INIT') THEN 'Init'
		WHEN UPPER(r.status) IN ('PICKED', 'PROCESSING') THEN 'Processing'
		ELSE r.status
		END AS refund_final_status

		,CASE
		WHEN UPPER(r.status) IN ('COMPLETED') THEN 'Completed'
		WHEN UPPER(r.status) IN ('CLONED', 'FAILED', 'CLONE_REQUESTED') THEN 'Failed'
		WHEN UPPER(r.status) IN ('CANCELLED') THEN 'Cancelled'
		WHEN UPPER(r.status) IN ('INIT','PICKED', 'PROCESSING') THEN 'Stuck'
		ELSE r.status
		END AS refund_final_status_updated

		,r.refund_mode
		,r.pg_id,
		case when r.pg_id in ('cfa_juspay_dmi_emi' ,'cfa_juspay_fibe_emi', 'cfa_juspay_abfl_emi', 'cfa_juspay_chola_emi', 'cfa_juspay_tvs_emi')
				then 'Yes' 
				else 'No'
				end as flipkart_emi_flag
		,r.bank_name
		,r.refund_gateway
		,r.refund_type
		,r.refund_reason
		,r.promise_date
		,refund_amount/100 as refund_amount
		,r.bank_reference_number


		,coalesce(f.auth_state, p2.auth_state) as auth_state
		,coalesce(p.sales_channel, f.transaction_source) as transaction_source

		--,Row_Number() over (partition by r.order_id,r.parent_refund_id order by r.child_updated_at Asc) as ref_asc_rank
		--,Row_Number() over (partition by r.order_id,r.parent_refund_id order by r.child_updated_at Desc) as ref_desc_rank

		,Row_Number() over (partition by r.order_id order by r.parent_created_at Asc) as ref_asc_rank_parent
		,Row_Number() over (partition by r.order_id order by r.parent_created_at Desc) as ref_desc_rank_parent
		,Row_Number() over (partition by r.order_id order by r.refund_amount Desc) as ref_amount_desc




		,COALESCE(p.instrument_type, f.instrument_type, p1.instrument_type, p2.instrument_type) as instrument_type
		,COALESCE(p.payment_instrument, f.payment_instrument, p1.payment_instrument, p2.payment_instrument) as payment_instrument
		,coalesce(p.payment_mode, p1.payment_mode) as payment_mode

	--From bigfoot_external_neo.cp_payments__fkpg_child_refund_base_history_fact as r
		From bigfoot_external_neo.cp_payments__fkpg_child_refund_base_fact as r

	Left Join (select transaction_id, instrument_type, payment_instrument,auth_state ,transaction_source from bigfoot_external_neo.cp_payments__fkpg_forward_transaction_base_history_fact 
		where date(transaction_time) >='2025-08-01') as f
		ON r.forward_transaction_id = f.transaction_id

		left join (select transaction_id, instrument_type, payment_instrument,payment_mode ,order_item_id ,sales_channel  from bigfoot_external_neo.cp_bi_prod_sales__forward_payments_365_fact 
		where unit_creation_date_key > 20250801) as p
		on r.forward_transaction_id = p.transaction_id

		LEFT JOIN (select transaction_id, instrument_type, payment_instrument,payment_mode from bigfoot_external_neo.cp_bi_prod_sales__forward_payments_fact 
		where unit_creation_date_key >= 20250801) AS p1
		ON r.forward_transaction_id = p1.transaction_id

		left join (select transaction_id, payment_instrument, instrument_type,auth_state from bigfoot_external_neo.cp_payments__fkpg_forward_transaction_base_fact 
		where transaction_date_key > 20250801) AS p2
		on r.forward_transaction_id = p2.transaction_id

		Where date(r.parent_created_at) >= "2025-04-01"
	) as ref
	Where ref_asc_rank_parent = 1
),

pymnt_srce as
(
	select checkout_id,
	count(distinct payment_type) as pymnt_cnt,
	max(payment_type) as payment_type_1
	
	from 
	(
		select checkout_Id,
		case when payment_instrument in ('FLIPKART_FINANCE', 'UPI_COLLECT', 'UPI_INTENT', 'DEBIT', 'CREDIT', 'EMI', 'RTGS', 'PHONEPE', 'DYNAMIC_QR', 'NET', 'FK_UPI', 'SUPERPAY_LATER','SUPERPAY_IN_INSTALLMENT','LOAN') then "Pre-paid"
		when payment_instrument in ('COD') then "COD"
		when payment_instrument in ('EGV', 'EGV_WALLET', 'WALLET', 'COIN') then "EGV/Wallet/Coin"
		else "Others" end as payment_type,
		count(distinct transaction_id) as txn_cnt
		
		--from bigfoot_external_neo.cp_payments__fkpg_forward_transaction_base_history_fact
		from bigfoot_external_neo.cp_payments__fkpg_forward_transaction_base_fact
		where  date(transaction_time) >= '2025-01-01'
		
		group by checkout_Id,
		case when payment_instrument in ('FLIPKART_FINANCE', 'UPI_COLLECT', 'UPI_INTENT', 'DEBIT', 'CREDIT', 'EMI', 'RTGS', 'PHONEPE', 'DYNAMIC_QR', 'NET', 'FK_UPI', 'SUPERPAY_LATER','SUPERPAY_IN_INSTALLMENT','LOAN') then "Pre-paid"
		when payment_instrument in ('COD') then "COD"
		when payment_instrument in ('EGV', 'EGV_WALLET', 'WALLET', 'COIN') then "EGV/Wallet/Coin"
		else "Others" end
	)
	
	group by checkout_id
),

lob_mapping_issue as
(
select contact_dt,
agent_id,
LOB,
case when y.FK_cnt is null or y.hyp_cnt is null then "NA"
	when (y.FK_cnt + y.hyp_cnt) = 0 then "NA"
	when y.FK_cnt / (y.FK_cnt + y.hyp_cnt) > 0.6 then "FK"
	else "Non-FK" end as FK_agent_flag from

(
	select completed_at as contact_dt,
	agent_id,
	LOB,
	count(distinct session_Id) as session_cnt,
	sum(HYP_flag) as hyp_cnt,
	sum(FK_flag) as FK_cnt
	
	from
	(
		select S.session_id,
		S.agent_id,
		S.completed_at,
		t.LOB,
		case when upper(o.marketplace_id) = "HYPERLOCAL" then 1 else 0 end as HYP_flag,
		case when upper(o.marketplace_id) = "FLIPKART" then 1 else 0 end as FK_flag
		
		from sessions S

		join inc I
		on S.incident_id = I.incident_id

		left join profiles t
		on upper(s.agent_id) = upper(t.ldap_id) and s.completed_at = t.day_level_snap

		left join orders O
		on I.order_external_id = O.order_external_id 
		--AND (I.order_item_id_list = 'null' or O.order_item_id = I.order_item_id_list)
	)x
	
	group by completed_at,
	agent_id,
	LOB
	)y
)
